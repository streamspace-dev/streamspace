Add Your Own Templates

Extend the catalog with custom application templates

YAML 
apiVersion: stream.space/v1alpha1
kind: Template
metadata:
  name: my-custom-app
  namespace: streamspace
spec:
  displayName: "My Custom Application"
  description: "Custom application template"
  category: "Custom"
  baseImage: "myregistry/myapp:latest"
  defaultResources:
    memory: 4Gi
    cpu: 2000m
  ports:
    - name: vnc
      containerPort: 5900
      protocol: TCP
  vnc:
    enabled: true
    port: 5900
  capabilities:
    - Network
    - Audio
    - Clipboard
  tags:
    - custom
    - productivity
